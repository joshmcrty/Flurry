This uses custom options to create a "confetti" effect: 

$('body').flurry({
  character: ["~"],
  color: ["#55476A", "#AE3D63", "#DB3853", "#F45C44", "#F8B646"],
  speed: 2000,
  height: 480,
  frequency: 60,
  small: 12,
  large: 50,
  rotation: 90,
  rotationVariance: 20,
  startRotation: 90,
  wind: 10,
  windVariance: 100,
  opacityEasing: "cubic-bezier(1,0,.96,.9)"
});
